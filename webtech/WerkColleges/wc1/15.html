Exercise 15

What is the output of the console after executing the code below?

function F(x){
  this.a = x;
}
F.prototype.foo = function(x) {
  return this.a + x;
}

var a = new F(2);
var b = new F(3);
console.log(a.foo(5));
console.log(b.foo(6));

F.prototype.foo = function(x) {
  return this.a - x;
}
console.log(a.foo(5));
console.log(b.foo(6));

a.foo = function(x) {
  return this.a * x;
}
console.log(a.foo(5));
console.log(b.foo(6));
