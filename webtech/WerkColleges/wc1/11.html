Exercise 11

What is the output of the console after executing the code below?

(hint: functions)

var a = 42;
var o = { name: 'Tom' }
var q = { name: 'Paul' }
function f(x, y, z) {
  x = -x;
  y.name = 'Diana';
  z = { name: 'Jennifer'};
}
f(a, o, q);

console.log(a);
console.log(o.name);
console.log(q.name);
